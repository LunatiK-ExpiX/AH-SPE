is_american_union = {
	main_parameter = nation
	this_parameter = nation
	
	exists = yes
	OR = { primary_culture = yankee primary_culture = dixie }
	is_greater_power = yes
	government = democracy
}

is_polish_union = {
	main_parameter = nation
	this_parameter = nation

	primary_culture = polish
	OR = { is_greater_power = yes is_secondary_power = yes }
}

is_influenced = {
	main_parameter = nation
	this_parameter = nation
	OR = {
		vassal_of = THIS
		substate_of = THIS
		in_sphere = THIS
	}
}

is_influenced = {
	main_parameter = province
	this_parameter = nation
	OR = {
		owned_by = THIS
		empty = yes
		owner = { test = is_influenced }
	}
}

has_usca_failed = {
	main_parameter = nation
	this_parameter = nation
	ELS = {
		exists = yes
		test != is_influenced
	}
	HON = {
		exists = yes
		test != is_influenced
	}
	NIC = {
		exists = yes
		test != is_influenced
	}
	COS = {
		exists = yes
		test != is_influenced
	}
}

is_warable_country = {
	main_parameter = nation
	this_parameter = nation

	THIS = { is_disarmed = no }
	is_our_vassal != THIS
	has_country_modifier != neutrality_modifier
	THIS = { has_country_modifier != neutrality_modifier }
	THIS = { has_country_modifier != no_more_war }
	has_country_modifier != war_focus
}

has_elections = {
	main_parameter = nation
	vote_franschise != none_voting
}

is_minority_culture = {
	main_parameter = pop
	is_primary_culture = no
	is_accepted_culture = no
}

has_citizen_rights = {
	main_parameter = pop
	this_parameter = nation
	OR = {
		AND = {
			THIS = { citizens_rights = primary_culture_only }
			is_primary_culture = yes
		}
		AND = {
			THIS = { citizens_rights = culture_voting_only }
			test != is_minority_culture
		}
		THIS = { citizens_rights = all_voting_pops }
	}
}

can_apply_for_statehood = {
	main_parameter = nation
	this_parameter = nation

	war = no
	is_independant = yes
	civilized = yes
	government = democracy
	THIS = { war = no }
	THIS = { is_independant = yes }
	THIS = { civilized = yes }
	THIS = { government = democracy }

	any_neighbor_country = { tag = THIS }
	is_greater_power = no
	is_secondary_power = no
	THIS = { is_greater_power = yes }
}

is_great_game_buffer = {
	main_parameter = nation
	this_parameter = nation
	
	exists = yes
	is_vassal = no
	is_greater_power = no
	is_secondary_power = no
	capital_scope = { continent = asia }
	OR = {
		neighbour = RUS
		any_neighbor_country = { vassal_of = RUS }
	}
	OR = {
		neighbour = ENG
		any_neighbor_country = { vassal_of = ENG }
	}
}

is_alaskan_seller = {
	main_parameter = nation
	war = no
	OR = {
		any_owned = { is_core = LSK }
		any_sphere_member = { tag = LSK }
		is_our_vassal = LSK
	}
}

eligible_for_alaskan_purchase = {
	main_parameter = nation

	test != is_alaskan_seller
	nationalism_n_imperialism = 1
	war = no
	has_country_flag != refused_alaska
	OR = {
		alliance_with = RUS
		neighbour = LSK
		any_owned = {
			is_core = LSK
			any_neighbor_province = {
				owner = {
					OR = {
						capital_scope = { continent = europe }
						is_greater_power = yes
						is_secondary_power = yes
					}
				}
			}
		}
	}
}

		}
		OR = {
			government = absolute_monarchy

			government = proletarian_dictatorship
		}
	}
}
