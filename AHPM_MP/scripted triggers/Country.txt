is_american_union = {
	main_parameter = nation
	this_parameter = nation
	
	exists = yes
	OR = { primary_culture = yankee primary_culture = dixie }
	is_greater_power = yes
	government = democracy
}

is_polish_union = {
	main_parameter = nation

	primary_culture = polish
	OR = { is_greater_power = yes is_secondary_power = yes }
}

	main_parameter = nation
	this_parameter = nation
	OR = {
		vassal_of = THIS

		in_sphere = THIS
	}
}

is_influenced = {
	main_parameter = province
	this_parameter = nation
	OR = {
		owned_by = THIS
		empty = yes
		owner = { test = is_influenced }
	}
}

has_usca_failed = {
	OR = {
		ELS = { exists = yes test != is_influenced }
		HON = { exists = yes test != is_influenced }
		NIC = { exists = yes test != is_influenced }
		COS = { exists = yes test != is_influenced }
	}
}

is_warable_country = {
	main_parameter = nation
	this_parameter = nation

	THIS = { is_disarmed = no }
	is_our_vassal != THIS
	has_country_modifier != neutrality_modifier
	THIS = { has_country_modifier != neutrality_modifier }
	THIS = { has_country_modifier != no_more_war }
	OR = {
		has_country_modifier != war_focus
		THIS = {
			has_country_flag != exclusive_acquire_all_cores_CB
			has_country_flag != exclusive_humiliate_CB
			has_country_flag != exclusive_liberate_country_CB
			has_country_flag != exclusive_make_puppet_CB
		}
	}
}
