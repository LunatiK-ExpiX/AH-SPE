# 1st Olympiad - Send Delegation or No
country_event = {
	id = 23403 # Attend 1896 Olympics
	title = "EVTNAME23403"
	desc = "EVTDESC23403"
	picture = "Olympics"
	is_triggered_only = yes
	option = {
		name = "EVTOPTA23403" # Send a delegation to the games
		set_country_flag = attend_olympiad
		prestige = 10
		if = {
			limit = {
				NOT = {
					any_country = {
						has_country_flag = ioc_member
						has_country_flag != attend_olympiad
						has_country_flag != noattend_olympiad
					}
				}
			}
			random_country = {
				limit = { has_country_flag = host_olympiad }
				country_event = 23404
			}
		}
		ai_chance = { factor = 1 }
	}
	option = {
		name = "EVTOPTB23403" # We're NOT going this year
		set_country_flag = noattend_olympiad
		prestige = -10
		ai_chance = { factor = 0 }
	}
}
